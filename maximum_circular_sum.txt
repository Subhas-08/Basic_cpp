/* KADANE'S ALGORITHM */
int kadane(int arr[],int n) 
{ 
    int max_so_far=0,sum=0; 
    for(int i=0;i<n;i++) 
    { 
        sum+=arr[i]; 
        if(sum<0) sum=0; 
        if(max_so_far<sum) max_so_far=sum; 
    } 
    return max_so_far; 
} 

int maxCircularSum(int arr[],int n) 
{ 
    int max_sum=kadane(arr,n); 
    int s=0,i; 
    for(i=0;i<n;i++) 
    { 
        s+=arr[i]; 
        arr[i]=-arr[i]; 
    } 
    s=s+kadane(arr,n);  
    return max(s,max_sum); 
}
